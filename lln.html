## the Law of Large Numbers:chi-square distribution;
##   population mean = 5
library(animation)
ani.options(interval = 0.1, nmax = 150)
lln.ani(FUN = function(n, mu) rchisq(n, df = mu), mu = 5, 
    cex = 0.6)

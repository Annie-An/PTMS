## CLT:normal distribution
library(animation)
ani.options(interval = 0.5, nmax = 150)
par(mar = c(3, 3, 0.2, 0.1), mgp = c(1.5, 0.5, 0), tcl = -0.3)
clt.ani(FUN = runif, mean = 0.5, sd = sqrt(1/12), type = "s")

## CLT:Chi-square distribution
library(animation)
ani.options(interval = 0.5, nmax = 150)
par(mar = c(3, 3, 1, 0.5), mgp = c(1.5, 0.5, 0), tcl = -0.3)
f = function(n) rchisq(n, 5)
clt.ani(FUN = f, mean = 5, sd = sqrt(2 * 5))
